import { describe, it, before, after } from 'node:test';
import assert from 'node:assert';
import { initializeTracing, shutdownTracing, resetSDKForTesting } from '../instrumentation.js';
import { withAgentSpan, withLLMSpan, withToolSpan } from '../span-helpers.js';
import { getTracer } from '@opentelemetry/api';
import { InMemorySpanExporter } from '@opentelemetry/sdk-trace-base';
import { SimpleSpanProcessor } from '@opentelemetry/sdk-trace-base';

describe('tracing/span-helpers', () => {
  const exporter = new InMemorySpanExporter();

  before(async () => {
    resetSDKForTesting();
    const sdk = await initializeTracing();
    sdk.addSpanProcessor(new SimpleSpanProcessor(exporter));
  });

  afterEach(() => {
    exporter.reset();
  });

  after(async () => {
    await shutdownTracing();
  });

  it('withAgentSpan creates span with correct attributes', async () => {
    await withAgentSpan('ds-20260326-001', 'test_action', async (span) => {
      span.addEvent('test_event');
    });

    const spans = exporter.getFinishedSpans();
    assert.strictEqual(spans.length, 1);
    assert.strictEqual(spans[0].name, 'agent.test_action');
    assert.strictEqual(spans[0].attributes['infogen.agent_action'], 'test_action');
  });

  it('withAgentSpan includes statement_id in attributes', async () => {
    await withAgentSpan('ds-20260326-001', 'test_action', async () => {
      // No-op
    });

    const spans = exporter.getFinishedSpans();
    assert.strictEqual(spans[0].attributes['infogen.statement_id'], 'ds-20260326-001');
  });

  it('withAgentSpan records exceptions on error', async () => {
    await assert.rejects(
      withAgentSpan('ds-20260326-001', 'failing_action', async () => {
        throw new Error('Test error');
      })
    );

    const spans = exporter.getFinishedSpans();
    assert.strictEqual(spans[0].status.code, 2); // ERROR
  });

  it('withLLMSpan creates nested span with LLM attributes', async () => {
    await withAgentSpan('ds-20260326-001', 'llm_wrapper', async () => {
      await withLLMSpan('anthropic', 'claude-3-sonnet', async (span) => {
        span.setAttribute('infogen.llm.input_tokens', 100);
        span.setAttribute('infogen.llm.output_tokens', 50);
      });
    });

    const spans = exporter.getFinishedSpans();
    const llmSpan = spans.find((s) => s.name === 'llm.inference');
    assert.ok(llmSpan, 'LLM span should exist');
    assert.strictEqual(llmSpan?.attributes['infogen.llm.provider'], 'anthropic');
    assert.strictEqual(llmSpan?.attributes['infogen.llm.model'], 'claude-3-sonnet');
  });

  it('withToolSpan creates nested span with tool attributes', async () => {
    await withAgentSpan('ds-20260326-001', 'tool_wrapper', async () => {
      await withToolSpan('write_file', { path: '/tmp/test.txt' }, async (span) => {
        span.setAttribute('infogen.tool.result', 'success');
      });
    });

    const spans = exporter.getFinishedSpans();
    const toolSpan = spans.find((s) => s.name === 'tool.write_file');
    assert.ok(toolSpan, 'Tool span should exist');
    assert.strictEqual(toolSpan?.attributes['infogen.tool.name'], 'write_file');
  });

  it('spans have millisecond-precision timestamps', async () => {
    await withAgentSpan('ds-20260326-001', 'timestamp_test', async () => {
      // No-op
    });

    const spans = exporter.getFinishedSpans();
    const span = spans[0];
    assert.ok(span.startTime, 'Span should have startTime');
    assert.ok(Array.isArray(span.startTime), 'startTime should be HrTime array [seconds, nanos]');
    assert.strictEqual(span.startTime.length, 2);
    // HrTime: [seconds, nanoseconds] provides nanosecond precision (> millisecond)
  });
});

import { TraceStorage, resetStorageForTesting } from '../trace-storage.js';

import { CUSTOM_ATTRIBUTES } from '../../tracing/attributes.js';

describe('storage/trace-storage', () => {
  let storage: TraceStorage;

  beforeEach(() => {
    resetStorageForTesting();
    storage = new TraceStorage({ maxTraces: 10 });
  });

  const createMockSpan = (traceId: string, name: string): ReadableSpan => {
    return {
      name,
      spanContext: () => ({ traceId, spanId: '12345678' }),
      startTime: [1000, 0],
      endTime: [1001, 0],
      duration: [1, 0],
      attributes: {
        [CUSTOM_ATTRIBUTES.STATEMENT_ID]: `ds-${traceId.substring(0, 8)}-001`,
      },
      status: { code: 0 },
      kind: 1,
      events: [],
      links: [],
    } as unknown as ReadableSpan;
  };

  it('stores spans grouped by traceId', () => {
    const span1 = createMockSpan('trace1', 'span1');
    const span2 = createMockSpan('trace1', 'span2');

    storage.store(span1);
    storage.store(span2);

    const trace = storage.get('trace1');
    assert.ok(trace);
    assert.strictEqual(trace.spans.length, 2);
  });

  it('get retrieves stored trace', () => {
    const span = createMockSpan('trace1', 'span1');
    storage.store(span);

    const trace = storage.get('trace1');
    assert.ok(trace);
    assert.strictEqual(trace.traceId, 'trace1');
  });

  it('getByStatementId finds traces by statement_id', () => {
    const span = createMockSpan('trace1', 'span1');
    storage.store(span);

    const traces = storage.getByStatementId('ds-trace1-001');
    assert.strictEqual(traces.length, 1);

    const span = createMockSpan('trace1', 'span1');

    storage.clear();

    assert.strictEqual(storage.getAll().length, 0);
  });

    storage.store(span);

    const filepath = await storage.exportToFile('trace1', 'test-trace.json');
    assert.ok(filepath.includes('test-trace.json'));
  });
});

#!/usr/bin/env node
/**
 * InfoGen Trace Capture Demo
 *
 * Demonstrates complete trace capture with hierarchical spans for agent execution.
 * Shows Agent → LLM and Agent → Tool span hierarchies.
 *
 * Run: node --import tsx src/examples/trace-demo.ts
 */

import { initializeTracing, shutdownTracing, withAgentSpan, withLLMSpan, withToolSpan } from '../tracing/index.js';
import { createConsoleExporter } from '../exporters/index.js';
import { SimpleSpanProcessor } from '@opentelemetry/sdk-trace-base';
import { storeTrace } from '../storage/index.js';

// Demo configuration
const DEMO_STATEMENT_ID = 'ds-20260326-001';
const DEMO_TASK = 'Build a login form with email and password fields';

/**
 * Mock LLM call (simulates Anthropic API)
 */
async function mockLLMCall(prompt: string): Promise<string> {
  // Simulate API latency
  await new Promise((resolve) => setTimeout(resolve, 100));

  return `Plan for: ${prompt}
1. Create HTML form structure
2. Add CSS styling
3. Implement validation
4. Add submit handler`;
}

/**
 * Mock tool call (simulates file write)
 */
async function mockToolCall(toolName: string, args: Record<string, unknown>): Promise<string> {
  await new Promise((resolve) => setTimeout(resolve, 50));

  if (toolName === 'write_file') {
    return `Wrote ${args.path} (${args.content?.length || 0} bytes)`;
  }
  return `Executed ${toolName}`;
}

/**
 * Main agent execution demo
 */
async function main() {
  console.log('=== InfoGen Trace Capture Demo ===\n');

  // Initialize tracing with console exporter for visibility
  const sdk = await initializeTracing({
    serviceName: 'infogen-demo',
    serviceVersion: '1.0.0-demo',
  });

  // Add console exporter to see traces in terminal
  const consoleExporter = createConsoleExporter({ indent: 2 });
  sdk.addSpanProcessor(new SimpleSpanProcessor(consoleExporter));

  console.log('Starting agent execution...\n');

  // Agent action: Execute task with LLM and tool calls
  await withAgentSpan(DEMO_STATEMENT_ID, 'execute_task', async (agentSpan) => {
    agentSpan.addEvent('task_started', { task: DEMO_TASK });

    // Step 1: LLM call for planning
    const plan = await withLLMSpan('anthropic', 'claude-3-sonnet-20240229', async (llmSpan) => {
      llmSpan.addEvent('llm_call_started');

      const response = mockLLMCall(DEMO_TASK);

      // Record token usage (simulated)
      llmSpan.setAttributes({
        'infogen.llm.input_tokens': 150,
        'infogen.llm.output_tokens': 80,
      });

      llmSpan.addEvent('llm_call_completed');
      return response;
    });

    console.log('LLM Plan:', plan);

    // Step 2: Tool call to write HTML file
    const htmlContent = `
<form>
  <input type="email" name="email" placeholder="Email" required />
  <input type="password" name="password" placeholder="Password" required />
  <button type="submit">Login</button>
</form>`;

    await withToolSpan('write_file', { path: './login-form.html', content: htmlContent }, async (toolSpan) => {
      const result = mockToolCall('write_file', { path: './login-form.html' });

      toolSpan.setAttribute('infogen.tool.result', 'success');
      toolSpan.addEvent('file_written', { path: './login-form.html', size: htmlContent.length });

      return result;
    });

    agentSpan.addEvent('task_completed');
    agentSpan.setAttribute('infogen.agent_status', 'success');
  });

  console.log('\nAgent execution complete.\n');

  // Shutdown and flush traces
  await shutdownTracing();

  console.log('Traces exported to console and OTLP endpoint.');
  console.log('Check ./traces/ for exported JSON files.');
}

// Run demo
main().catch((error) => {
  console.error('Demo failed:', error);
  process.exit(1);
});

# InfoGen Trace Infrastructure

OpenTelemetry-compatible trace capture for agent execution observability.

## Overview

This module provides hierarchical span tracing for AI agent operations, enabling observability into LLM calls, tool executions, and agent actions. Traces are exported in OTLP format for compatibility with standard observability backends (Jaeger, Grafana, Datadog, etc.).

## Features

- **OpenTelemetry Compatible**: Uses standard OTLP export format
- **Hierarchical Spans**: Agent → LLM/Tool span nesting
- **Phase 1 Integration**: Includes `statement_id` for traceability
- **Millisecond Precision**: Nanosecond-precision timestamps via HrTime
- **Flexible Exporters**: OTLP (production), Console (development)
- **In-Memory Storage**: Store and export traces to JSON files

## Installation

\`\`\`bash
npm install @opentelemetry/sdk-node @opentelemetry/api @opentelemetry/exporter-trace-otlp-proto
\`\`\`

## Quick Start

\`\`\`typescript
import { initializeTracing, withAgentSpan, withLLMSpan, withToolSpan } from './tracing/index.js';

// Initialize at application startup
await initializeTracing();

// Use in agent code
await withAgentSpan('ds-20260326-001', 'execute_task', async (span) => {
  span.addEvent('task_started');

  // LLM call (nested span)
  const plan = await withLLMSpan('anthropic', 'claude-3-sonnet', async (llmSpan) => {
    const response = await llmClient.messages.create({ model: 'claude-3-sonnet', messages });
    llmSpan.setAttributes({
      'infogen.llm.input_tokens': response.usage.input_tokens,
      'infogen.llm.output_tokens': response.usage.output_tokens,
    });
    return response;
  });

  // Tool call (nested span)
  await withToolSpan('write_file', { path: '/tmp/output.txt' }, async (toolSpan) => {
    await fs.writeFile('/tmp/output.txt', content);
    toolSpan.setAttribute('infogen.tool.result', 'success');
  });

  span.addEvent('task_completed');
});

## API Reference

### Initialization

\`\`\`typescript
import { initializeTracing, shutdownTracing } from './tracing/index.js';

// Initialize with optional config
await initializeTracing({
  serviceName: 'my-agent',
  tracesEndpoint: 'http://localhost:4318/v1/traces',
});

// Graceful shutdown
await shutdownTracing();
\`\`\`

### Span Helpers

\`\`\`typescript
import { withAgentSpan, withLLMSpan, withToolSpan } from './tracing/index.js';

// Agent action span (top-level)
await withAgentSpan(statementId, actionName, async (span) => {
  // Agent logic here
});

// LLM call span (nested)
await withLLMSpan(provider, model, async (span) => {
  // LLM call here
});

// Tool call span (nested)
await withToolSpan(toolName, args, async (span) => {
  // Tool execution here
});
\`\`\`

### Attributes

\`\`\`typescript
import { CUSTOM_ATTRIBUTES, addTraceability } from './tracing/index.js';

// Add Phase 1 traceability
addTraceability(span, 'ds-20260326-001', {
  deliverableType: 'code/framework',
  logLevel: 'INFO',
});
