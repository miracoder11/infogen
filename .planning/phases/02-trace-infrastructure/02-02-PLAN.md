import { Span } from '@opentelemetry/api';

/**
 * Custom attribute names for InfoGen trace capture.
 * These integrate Phase 1 traceability requirements with OpenTelemetry.
 *
 * Prefix convention: 'infogen.' avoids conflicts with semantic conventions.
 *
 * @see docs/METHODOLOGY.md Section 3.5 - Traceability Requirements
 */
export const CUSTOM_ATTRIBUTES = {
  // Phase 1 traceability attributes
  /** Demo statement ID: ds-{YYYYMMDD}-{sequence} */
  STATEMENT_ID: 'infogen.statement_id',

  /** Deliverable type: code, methodology, test, change, or custom/{name} */
  DELIVERABLE_TYPE: 'infogen.deliverable_type',

  /** Log level: DEBUG, INFO, WARN, ERROR, PASS, FAIL (from Phase 1) */
  LOG_LEVEL: 'infogen.log_level',

  // Agent-specific attributes
  /** Agent action name (e.g., 'execute_task', 'plan_action') */
  AGENT_ACTION: 'infogen.agent_action',

  /** Agent execution status: running, success, failed */
  AGENT_STATUS: 'infogen.agent_status',

  /** Tool name used in agent execution */
  TOOL_NAME: 'infogen.tool.name',

  /** Tool call arguments (JSON stringified) */
  TOOL_ARGUMENTS: 'infogen.tool.arguments',

  /** Tool execution result */
  TOOL_RESULT: 'infogen.tool.result',

  // LLM-specific attributes
  /** LLM provider (e.g., 'anthropic', 'openai') */
  LLM_PROVIDER: 'infogen.llm.provider',

  /** LLM model identifier */
  LLM_MODEL: 'infogen.llm.model',

  /** Input token count */
  LLM_INPUT_TOKENS: 'infogen.llm.input_tokens',

  /** Output token count */
  LLM_OUTPUT_TOKENS: 'infogen.llm.output_tokens',
} as const;

/**
 * Add Phase 1 traceability attributes to any span.
 *
 * @param span - The span to enrich with attributes
 * @param statementId - Demo statement ID (ds-{YYYYMMDD}-{sequence})
 * @param metadata - Optional additional metadata
 */
export function addTraceability(
  span: Span,
  statementId: string,
  metadata?: {
    deliverableType?: string;
    logLevel?: keyof typeof LOG_LEVELS;
  }
): void {
  const attributes: Record<string, string | number> = {
    [CUSTOM_ATTRIBUTES.STATEMENT_ID]: statementId,
  };

  if (metadata?.deliverableType) {
    attributes[CUSTOM_ATTRIBUTES.DELIVERABLE_TYPE] = metadata.deliverableType;
  }

  if (metadata?.logLevel) {
    attributes[CUSTOM_ATTRIBUTES.LOG_LEVEL] = metadata.logLevel;
  }

  span.setAttributes(attributes);
}

/**
 * Log level constants matching Phase 1 JSON Lines format.
 */
export const LOG_LEVELS = {
  DEBUG: 'DEBUG',
  INFO: 'INFO',
  WARN: 'WARN',
  ERROR: 'ERROR',
  PASS: 'PASS',
  FAIL: 'FAIL',
} as const;

/**
 * Agent status constants.
 */
export const AGENT_STATUS = {
  RUNNING: 'running',
  SUCCESS: 'success',
  FAILED: 'failed',
} as const;

/**
 * Deliverable type constants from Phase 1.
 */
export const DELIVERABLE_TYPES = {
  CODE_FRAMEWORK: 'code/framework',
  METHODOLOGY_SKILL: 'methodology/skill',
  TEST_QUALITY: 'test/quality',
  CHANGE_REFACTOR: 'change/refactor',
} as const;

```typescript
import { trace, Span, SpanStatusCode } from '@opentelemetry/api';
import { CUSTOM_ATTRIBUTES, AGENT_STATUS, addTraceability } from './attributes.js';

/** Tracer name for InfoGen agent operations. */
const TRACER_NAME = 'infogen-agent';
/** Tracer version. */
const TRACER_VERSION = '1.0.0';

/**
 * Get the global tracer instance.
 * Must be called after SDK initialization via initializeTracing().
 */
function getTracer() {
  return trace.getTracer(TRACER_NAME, TRACER_VERSION);
}

/**
 * Wrap an agent action with a span.
 *
 * Creates a top-level span representing a single agent action.
 * Automatically adds traceability attributes and error handling.
 *
 * @example
 * ```ts
 * const result = await withAgentSpan('ds-20260326-001', 'execute_task', async (span) => {
 *   span.addEvent('task_started', { task: 'Build login form' });
 *   // ... agent logic
 *   return { success: true };
 * });
 * ```
 */
export async function withAgentSpan<T>(
  statementId: string,
  actionName: string,
  fn: (span: Span) => Promise<T>
): Promise<T> {
  const tracer = getTracer();

  return tracer.startActiveSpan(
    `agent.${actionName}`,
    {
      attributes: {
        [CUSTOM_ATTRIBUTES.AGENT_ACTION]: actionName,
        [CUSTOM_ATTRIBUTES.AGENT_STATUS]: AGENT_STATUS.RUNNING,
      },
    },
    async (span) => {
      // Add Phase 1 traceability
      addTraceability(span, statementId);

      try {
        const result = await fn(span);
        span.setStatus({ code: SpanStatusCode.OK });
        span.setAttribute(CUSTOM_ATTRIBUTES.AGENT_STATUS, AGENT_STATUS.SUCCESS);
        return result;
      } catch (error) {
        const message = error instanceof Error ? error.message : String(error);
        span.recordException(error);
        span.setStatus({
          code: SpanStatusCode.ERROR,
          message,
        });
        span.setAttribute(CUSTOM_ATTRIBUTES.AGENT_STATUS, AGENT_STATUS.FAILED);
        throw error;
      } finally {
        span.end();
      }
    }
  );
}

/**
 * Wrap an LLM inference call with a span.
 *
 * Creates a nested span for LLM operations. Uses GenAI semantic conventions
 * where available, with custom attributes for InfoGen-specific fields.
 *
 * Should be called within an agent span to maintain hierarchy.
 *
 * @example
 * ```ts
 * await withAgentSpan('ds-20260326-001', 'chat', async (agentSpan) => {
 *   const response = await withLLMSpan('anthropic', 'claude-3-sonnet-20240229', async (llmSpan) => {
 *     // ... LLM call
 *     llmSpan.setAttributes({
 *       [CUSTOM_ATTRIBUTES.LLM_INPUT_TOKENS]: response.usage.input_tokens,
 *       [CUSTOM_ATTRIBUTES.LLM_OUTPUT_TOKENS]: response.usage.output_tokens,
 *     });
 *     return response;
 *   });
 * });
 * ```
 */
export async function withLLMSpan<T>(
  provider: string,
  model: string,
  fn: (span: Span) => Promise<T>
): Promise<T> {
  const tracer = getTracer();

  return tracer.startActiveSpan(
    'llm.inference',
    {
      attributes: {
        [CUSTOM_ATTRIBUTES.LLM_PROVIDER]: provider,
        [CUSTOM_ATTRIBUTES.LLM_MODEL]: model,
        // GenAI semantic conventions (experimental as of 2026-03)
        'gen_ai.operation.name': 'chat',
        'gen_ai.request.model': model,
        'gen_ai.provider': provider,
      },
    },
    async (span) => {
      try {
        const result = await fn(span);
        span.setStatus({ code: SpanStatusCode.OK });
        return result;
      } catch (error) {
        const message = error instanceof Error ? error.message : String(error);
        span.recordException(error);
        span.setStatus({
          code: SpanStatusCode.ERROR,
          message,
        });
        throw error;
      } finally {
        span.end();
      }
    }
  );
}

/**
 * Wrap a tool call with a span.
 *
 * Creates a nested span for tool/function calls. Records tool name,
 * arguments, and result for observability.
 *
 * Should be called within an agent span to maintain hierarchy.
 *
 * @example
 * ```ts
 * await withAgentSpan('ds-20260326-001', 'use_tools', async (agentSpan) => {
 *   const result = await withToolSpan('write_file', { path: '/tmp/output.txt' }, async (toolSpan) => {
 *     const content = await fs.writeFile('/tmp/output.txt', 'Hello, World!');
 *     toolSpan.setAttribute(CUSTOM_ATTRIBUTES.TOOL_RESULT, 'success');
 *     return content;
 *   });
 * });
 * ```
 */
export async function withToolSpan<T>(
  toolName: string,
  args: Record<string, unknown>,
  fn: (span: Span) => Promise<T>
): Promise<T> {
  const tracer = getTracer();

  return tracer.startActiveSpan(
    `tool.${toolName}`,
    {
      attributes: {
        [CUSTOM_ATTRIBUTES.TOOL_NAME]: toolName,
        [CUSTOM_ATTRIBUTES.TOOL_ARGUMENTS]: JSON.stringify(args),
        // GenAI semantic conventions
        'gen_ai.operation.name': 'execute_tool',
        'gen_ai.tool.name': toolName,
        'gen_ai.tool.call.arguments': JSON.stringify(args),
      },
    },
    async (span) => {
      try {
        const result = await fn(span);
        span.setStatus({ code: SpanStatusCode.OK });
        return result;
      } catch (error) {
        const message = error instanceof Error ? error.message : String(error);
        span.recordException(error);
        span.setStatus({
          code: SpanStatusCode.ERROR,
          message,
        });
        throw error;
      } finally {
        span.end();
      }
    }
  );
}

/**
 * Create a manual child span without automatic wrapping.
 *
 * Use this for advanced scenarios where you need more control over
 * span lifecycle. Remember to call span.end() when done.
 *
 * @example
 * ```ts
 * const span = createChildSpan('custom_operation');
 * try {
 *   // ... custom logic
 *   span.setStatus({ code: SpanStatusCode.OK });
 * } finally {
 *   span.end();
 * }
 * ```
 */
export function createChildSpan(name: string, attributes?: Record<string, unknown>): Span {
  const tracer = getTracer();
  const span = tracer.startSpan(name, { attributes });
  return span;
}

import { NodeSDK } from '@opentelemetry/sdk-node';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-proto';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';
import { BatchSpanProcessor } from '@opentelemetry/sdk-trace-base';

/**
 * OpenTelemetry SDK instance.
 * Initialized once at application startup.
 */
let sdk: NodeSDK | null = null;

/**
 * Configuration options for trace initialization.
 */
export interface TracingConfig {
  /** Service name (appears in trace backend) */
  serviceName?: string;
  /** Service version */
  serviceVersion?: string;
  /** OTLP traces endpoint URL */
  tracesEndpoint?: string;
  /** OTLP authentication token (optional) */
  authToken?: string;
  /** Export timeout in milliseconds (default: 10000) */
  timeoutMillis?: number;
  /** Sampling rate 0.0 to 1.0 (default: 1.0) */
  samplingRate?: number;
}

/**
 * Initialize OpenTelemetry SDK for trace capture.
 *
 * Must be called once at application startup, before any tracing operations.
 * Registers shutdown handler for graceful exit.
 *
 * @example
 * ```ts
 * import { initializeTracing } from './tracing/instrumentation.js';
 *
 * await initializeTracing({
 *   serviceName: 'infogen-agent',
 *   tracesEndpoint: 'http://localhost:4318/v1/traces',
 * });
 * ```
 *
 * @param config - Optional configuration overrides
 * @returns The initialized SDK instance
 */
export async function initializeTracing(config: TracingConfig = {}): Promise<NodeSDK> {
  if (sdk) {
    console.warn('[InfoGen Tracing] SDK already initialized, returning existing instance');
    return sdk;
  }

  const {
    serviceName = process.env.OTEL_SERVICE_NAME || 'infogen-agent',
    serviceVersion = process.env.OTEL_SERVICE_VERSION || '1.0.0',
    tracesEndpoint = process.env.OTEL_EXPORTER_OTLP_TRACES_ENDPOINT || 'http://localhost:4318/v1/traces',
    authToken = process.env.OTEL_AUTH_TOKEN,
    timeoutMillis = parseInt(process.env.OTEL_EXPORTER_OTLP_TIMEOUT || '10000', 10),
    samplingRate = parseFloat(process.env.OTEL_SAMPLING_RATE || '1.0'),
  } = config;

  // Create OTLP trace exporter
  const traceExporter = new OTLPTraceExporter({
    url: tracesEndpoint,
    headers: authToken ? { Authorization: `Bearer ${authToken}` } : undefined,
    timeoutMillis,
  });

  // Create resource with service identification
  const resource = new Resource({
    [SemanticResourceAttributes.SERVICE_NAME]: serviceName,
    [SemanticResourceAttributes.SERVICE_VERSION]: serviceVersion,
    // Add custom attributes
    'infogen.project.name': 'infogen',
    'infogen.trace.version': '1.0.0',
  });

  // Initialize SDK
  sdk = new NodeSDK({
    resource,
    traceExporter,
    // BatchSpanProcessor is default, handles batching and retries
    // spanProcessor: new BatchSpanProcessor(traceExporter),
  });

  // Start SDK
  await sdk.start();
  console.log(`[InfoGen Tracing] SDK initialized: ${serviceName} v${serviceVersion}`);
  console.log(`[InfoGen Tracing] Exporting to: ${tracesEndpoint}`);

  // Register shutdown handlers
  registerShutdown(sdk);

  return sdk;
}

/**
 * Shutdown OpenTelemetry SDK gracefully.
 *
 * Flushes all pending spans and closes connections.
 * Should be called on application exit.
 */
export async function shutdownTracing(): Promise<void> {
  if (!sdk) {
    console.warn('[InfoGen Tracing] SDK not initialized, nothing to shutdown');
    return;
  }

  await sdk.shutdown();
  console.log('[InfoGen Tracing] SDK shut down');
  sdk = null;
}

/**
 * Check if tracing SDK is initialized.
 */
export function isTracingInitialized(): boolean {
  return sdk !== null;
}

/**
 * Register shutdown handlers for graceful exit.
 */
function registerShutdown(sdkInstance: NodeSDK): void {
  const shutdownHandler = async (signal: string) => {
    console.log(`[InfoGen Tracing] Received ${signal}, shutting down...`);
    await sdkInstance.shutdown();
    process.exit(0);
  };

  process.on('SIGTERM', () => shutdownHandler('SIGTERM'));
  process.on('SIGINT', () => shutdownHandler('SIGINT'));
}

/**
 * Reset SDK instance (for testing purposes).
 *
 * @internal
 */
export function resetSDKForTesting(): void {
  sdk = null;
}
