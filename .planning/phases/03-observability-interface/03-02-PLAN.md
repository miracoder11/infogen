### Task 2: Timeline Component
**type: "auto" tdd: "true"**

Create Timeline visualization component:

**behavior:**
- Renders spans as horizontal bars
- Bar length proportional to duration
- Nested spans shown at increasing depths
- Color-coded by status (green=ok, red=error)
- Hover shows span details tooltip
- Click selects span for detail view

**implementation:**
```tsx
// ui/src/components/Timeline.tsx
export function Timeline({ traceId }: { traceId: string }) {
  const [spans, setSpans] = useState<TimelineSpan[]>([]);
  const [selected, setSelected] = useState<string | null>(null);
  const [zoom, setZoom] = useState(1);

  return (
    <div className="timeline-container">
      <div className="timeline-controls">
        <button onClick={() => setZoom(z => z * 1.5)}>Zoom In</button>
        <button onClick={() => setZoom(z => z / 1.5)}>Zoom Out</button>
      </div>
      <div className="timeline">
        {spans.map(span => (
          <SpanBar
            key={span.id}
            span={span}
            zoom={zoom}
            selected={selected === span.id}
            onClick={() => setSelected(span.id)}
          />
        ))}
      </div>
    </div>
  );
}
```
