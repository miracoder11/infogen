### Task 1: Span Content Inspector
**type: "auto"**

Create inspector component to show span attributes and content:

```tsx
// ui/src/components/SpanInspector.tsx
import ReactMarkdown from 'react-markdown';
import { Prism as SyntaxHighlighter } from 'react-syntax-highlighter';

export function SpanInspector({ span }: { span: TimelineSpan }) {
  const prompt = span.attributes['infogen.llm.prompt'];
  const response = span.attributes['infogen.llm.response'];

  return (
    <div className="span-inspector">
      <h3>{span.name}</h3>
      <div className="attributes">
        {Object.entries(span.attributes).map(([k, v]) => (
          <div key={k}><strong>{k}:</strong> {String(v)}</div>
        ))}
      </div>
      {prompt && (
        <div className="prompt">
          <h4>Prompt</h4>
          <ReactMarkdown>{String(prompt)}</ReactMarkdown>
        </div>
      )}
      {response && (
        <div className="response">
          <h4>Response</h4>
          <SyntaxHighlighter language="markdown">
            {String(response)}
          </SyntaxHighlighter>
        </div>
      )}
    </div>
  );
}
```

**done:**
- SpanInspector shows all attributes
- Markdown content rendered
- Code blocks syntax highlighted

### Task 3: Search Functionality
**type: "auto"**

Add search across trace content:

```tsx
// ui/src/components/SearchBar.tsx
export function SearchBar({ onSearch }: { onSearch: (query: string) => void }) {
  const [query, setQuery] = useState('');

  const debouncedSearch = useMemo(
    () => debounce((q: string) => onSearch(q), 300),
    [onSearch]
  );

  return (
    <input
      type="search"
      placeholder="Search traces..."
      value={query}
      onChange={e => { setQuery(e.target.value); debouncedSearch(e.target.value); }}
    />
  );
}
```

API endpoint enhancement:

```typescript
// src/api/server.ts - add search
app.get('/api/traces', (req, res) => {
  const { q, statementId, status, since } = req.query;
  let traces = getTraceStorage().getAll();

  if (q) traces = searchTraces(traces, String(q));
  if (statementId) traces = traces.filter(t => t.statementId === statementId);
  if (status) traces = traces.filter(t => t.spans.some(s => s.status.code === (status === 'error' ? 2 : 1)));
  if (since) traces = traces.filter(t => t.startTime > Number(since));

  res.json(traces);
});
```
